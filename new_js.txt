  function triggerCompleteEffects(taskIds) {
    if (!Array.isArray(taskIds)) taskIds = [taskIds];
    
    if (typeof confetti !== 'undefined') {
      confetti({ particleCount: 100 * taskIds.length, spread: 70, origin: { y: 0.6 } });
    }
    playChimeSound();
    
    const rows = document.querySelectorAll('.dashboard-task-row');
    let maxWait = 0;

    taskIds.forEach(taskId => {
      let row = null;
      let taskText = 'Task';
      
      rows.forEach(r => {
        const cb = r.querySelector('.row-checkbox');
        if (cb && cb.value === taskId) {
          row = r;
          const clientCell = r.querySelector('td:nth-child(2)');
          const taskCell = r.querySelector('td:nth-child(3)');
          const clientName = clientCell ? clientCell.innerText.trim() : '';
          const taskName = taskCell ? taskCell.innerText.trim() : '';
          if (clientName && taskName) taskText = \\ - \\;
          else if (taskName) taskText = taskName;
        }
      });

      if (row) {
        row.classList.add('task-complete-anim');
        
        if (globalArchiveRule === '0') {
          maxWait = Math.max(maxWait, 5500);
          setTimeout(() => {
            row.classList.remove('task-complete-anim');
            row.classList.add('task-complete-anim-fadeout');
            setTimeout(() => {
              row.style.display = 'none'; // hide it
              fetch(\/tasks/\/archive-immediate\, { method: 'POST' });
            }, 500);
          }, 5000);
        } else {
          maxWait = Math.max(maxWait, 2500);
          setTimeout(() => {
            row.classList.remove('task-complete-anim');
            row.classList.add('task-complete-anim-fadeout');
            setTimeout(() => {
              row.classList.remove('task-complete-anim-fadeout');
            }, 2000);
          }, 500);
        }
      }
    });

    setTimeout(() => {
      if (globalArchiveRule === '0') showToast(taskIds.length > 1 ? \\ tasks moved to archives\ : 'Task moved to archives', 'success');
      else showToast(taskIds.length > 1 ? \\ tasks marked completed!\ : 'Task status updated!', 'success');
      refreshDashboardTable();
    }, maxWait);
  }

  async function quickMarkComplete(taskIds) {
    if (!Array.isArray(taskIds)) taskIds = [taskIds];
    
    const tasksToComplete = [];
    const tasksToRevert = [];
    
    taskIds.forEach(id => {
      const task = allTasks.find(t => t.id === id);
      if (task) {
        if (task.status === 'Completed') tasksToRevert.push(task);
        else tasksToComplete.push(task);
      }
    });

    if (tasksToComplete.length === 0 && tasksToRevert.length === 0) return;

    const updateTask = async (task, newStatus) => {
      const payload = {
        status: newStatus,
        client_id: task.client_id || '',
        task_type_id: task.task_type_id || '',
        assigned_user_id: task.assigned_user_id || '',
        due_date: task.due_date || '',
        tags: (task.tags || []).join(', ')
      };
      const response = await fetch('/tasks/' + task.id, {
        method: 'PUT',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(payload)
      });
      return response.ok;
    };

    const promises = [];
    tasksToComplete.forEach(t => promises.push(updateTask(t, 'Completed')));
    tasksToRevert.forEach(t => promises.push(updateTask(t, 'In Progress')));
    
    await Promise.all(promises);

    if (tasksToComplete.length > 0) {
      triggerCompleteEffects(tasksToComplete.map(t => t.id));
    } else if (tasksToRevert.length > 0) {
      showToast('Task statuses reverted!', 'success');
      refreshDashboardTable();
    }
  }
